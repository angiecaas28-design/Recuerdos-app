<script>
// Registrar Service Worker
if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('service-worker.js');
}

const fechaInicio = new Date("2025-02-28");
const contraseñaCorrecta = "MARAMOR";	
let recuerdos = JSON.parse(localStorage.getItem("recuerdos")) || [];

function checkPassword(){
  const pass = document.getElementById("password").value;
  if(pass === contraseñaCorrecta){
    document.getElementById("login").classList.add("hidden");
    document.getElementById("main").classList.remove("hidden");
    actualizarContador();
    mostrarRecuerdos();
  } else { alert("Contraseña incorrecta 💔"); }
}

function actualizarContador(){
  const hoy = new Date();
  const diff = hoy - fechaInicio;
  const dias = Math.floor(diff / (1000*60*60*24));
  document.getElementById("contador").innerText = "💖 " + dias + " días juntos";
}

function guardarLocal(){ localStorage.setItem("recuerdos", JSON.stringify(recuerdos)); }

function mostrarRecuerdos(){
  const div = document.getElementById("recuerdos");
  div.innerHTML = "";
  recuerdos.forEach(r => {
    const card = document.createElement("div");
    card.className = "card";
    card.innerHTML = `<h4>💗 ${r.titulo}</h4><p>${r.mensaje}</p>`;
    if(r.foto){
      const img = document.createElement("img");
      img.src = r.foto;
      card.appendChild(img);
    }
    div.appendChild(card);
  });
}

function agregarRecuerdo(){
  const titulo = document.getElementById("titulo").value;
  const mensaje = document.getElementById("mensaje").value;
  const fotoInput = document.getElementById("foto");
  if(titulo && mensaje){
    const nuevo = { titulo, mensaje, foto:null };
    if(fotoInput.files[0]){
      const reader = new FileReader();
      reader.onload = function(e){
        nuevo.foto = e.target.result;
        recuerdos.push(nuevo);
        guardarLocal();
        mostrarRecuerdos();
      }
      reader.readAsDataURL(fotoInput.files[0]);
    } else {
      recuerdos.push(nuevo);
      guardarLocal();
      mostrarRecuerdos();
    }
    document.getElementById("titulo").value="";
    document.getElementById("mensaje").value="";
    document.getElementById("foto").value="";
  }
}

function modoAniversario(){
  alert("🎉 Feliz aniversario mi amor 💖 Gracias por cada momento juntas");
}

const CACHE_NAME = 'recuerdos-cache-v1';
const urlsToCache = ['recuerdos.html'];

self.addEventListener('install', event => {
  event.waitUntil(
    caches.open(CACHE_NAME)
      .then(cache => cache.addAll(urlsToCache))
  );
});

self.addEventListener('fetch', event => {
  event.respondWith(
    caches.match(event.request)
      .then(response => response || fetch(event.request))
  );
});
